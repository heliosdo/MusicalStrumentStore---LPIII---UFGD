CREATE TABLE AMIGO (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    APELIDO VARCHAR(50),
    CIDADE VARCHAR(100),
    DATANASCIMENTO VARCHAR(20),
    EMAIL VARCHAR(100),
    ESTADOCIVIL VARCHAR(20),
    FACEBOOK VARCHAR(100),
    INSTAGRAM VARCHAR(100),
    NOME VARCHAR(100),
    SEXO VARCHAR(20),
    WHATSAPP VARCHAR(15)
);
